\section{Deferred Constructions}\label{app:deferred-constructions}

This appendix collects the technical details deferred from the main text. Each
subsection below records the statement to be proved, the expected witness
objects, the proof shape, and the soundness/completeness points that should be
filled in.

\subsection{Batched Table Lookup}\label{app:batched-lookup}

\paragraph{Goal.}
Give the concrete reduction from many shaped lookup relations
$\mathbf{F}_i\subseteq\mathbf{T}_i$ to one lookup instance, and then explain how
same-form lookup checks are batched by interleaving.

\paragraph{Framework to fill in.}
\begin{itemize}
    \item Define padding from width $m_i$ to a common width $m_{\max}$ and state
    how dummy columns are fixed.
    \item Define the random linear combination of rows under a verifier
    challenge and the resulting single-table lookup relation.
    \item State the collision/soundness bound from the random combination.
    \item Describe how the resulting single-column/basic lookup is instantiated
    using the lookup argument of~\cite{cq++24} and logup~\cite{logup22}.
    \item Explain how interleaving batching merges multiple lookup instances
    without new commitments once their local constraints have the same form.
\end{itemize}

\subsection{Prefix-Sum Relations}\label{app:prefix-sum}

\paragraph{Goal.}
Provide concrete PCS checks for the three prefix-sum forms used in
\Cref{sec:zkgbdt-protocol}: prefix along the last, middle, and first tensor
dimension.

\paragraph{Framework to fill in.}
\begin{itemize}
    \item State the tensor flattening convention and the ambient domain
    $\mathbb{H}_{abc}$.
    \item For each prefix direction, define the committed prefix tensor
    $\mathbf{g}$ and total tensor $\mathbf{h}$.
    \item Write the local recurrence identities for adjacent entries and the
    boundary identities that bind the first prefix value and the total value.
    \item Convert each recurrence and boundary identity into zero-checks over
    the appropriate domain.
    \item State how these linear checks are batched by domain-lifting batching.
\end{itemize}

\subsection{Replication Relations}\label{app:replication}

\paragraph{Goal.}
Give the concrete proof for row- and column-replication relations used to align
objects before lookup, range checks, histogram addressing, and internal-node
masks.

\paragraph{Framework to fill in.}
\begin{itemize}
    \item State the row-replication relation and its polynomial identity, e.g.,
    when replication can be represented by a substitution $X\mapsto X^r$.
    \item State the column-replication relation from
    $\mathbf{f}\in\mathbb{F}^n$ to
    $\mathbf{G}\in\mathbb{F}^{n\times m}$.
    \item Give the zero-check identities proving that every replicated row or
    column equals its source entry.
    \item Record any domain-size assumptions needed for nested domains.
    \item Explain how these linear identities join the domain-lifting batch.
\end{itemize}

\begin{algorithm}[H]
\caption{Quantized multi-class GBDT training trace}
\label{alg:gbdt-trace}
\begin{algorithmic}
\Require Quantized dataset $\mathbf{D}\in[B]^{M\times d}$, one-hot labels $\mathbf{Y}\in\{0,1\}^{K\times M}$, rounds $T$, learning rate $\mu$
\Ensure Forest metadata, leaf values, and proof-facing round trace
\State Set virtual initial scores $\mathbf{S}[-1,k,r]\gets 0$ for all $k,r$
\State Initialize $\mathbf{G}[0,k,r]\gets \mathbf{Y}[k,r]$ for all $k,r$
\For{$t=0$ to $T-1$}
  \For{$k=0$ to $K-1$}
    \State Infer global leaf indices $\boldsymbol{\ell}^{(t,k)}$ for all samples under the current tree
    \State Build leaf-supported count and weighted histograms from $\boldsymbol{\ell}^{(t,k)}$, $\mathbf{D}$, and $\mathbf{G}[t,k,\cdot]$
    \State Propagate histograms to internal nodes using the committed child index vectors
    \State Compute prefix statistics and candidate gain tensor $\Gamma^{(t,k)}$
    \State Select the maximizing feature-threshold pair at each internal node
    \State Compute padded leaf values $\widetilde{\mathbf{W}}^{(t)}[\cdot,k]$ from per-leaf counts and residual sums
  \EndFor
  \For{$r=0$ to $M-1$}
    \For{$k=0$ to $K-1$}
      \State $\mathbf{S}[t,k,r] \gets \mathbf{S}[t-1,k,r] + \mu\cdot \widetilde{\mathbf{W}}^{(t)}[\boldsymbol{\ell}^{(t,k)}[r],k]$
    \EndFor
  \EndFor
  \If{$t<T-1$}
    \State Compute $\mathbf{G}[t+1,\cdot,\cdot]$ from $\mathbf{S}[t,\cdot,\cdot]$ and $\mathbf{Y}$
  \EndIf
\EndFor
\State \Return committed forest, leaf values, and trace
\end{algorithmic}
\end{algorithm}

\begin{algorithm}[H]
\caption{Proof generation for quantized ZK-GBDT training}
\label{alg:zkgbdt-proof}
\begin{algorithmic}
\Require Commitments to the forest, dataset, labels, and the training trace of \cref{alg:gbdt-trace}
\Ensure Proof of correct batched inference and training
\State Prove forest well-formedness as in \cref{prop:forest-well-formedness}
\For{$t=0$ to $T-1$}
  \State Prove boosting-state and pseudo-residual correctness as in \cref{prop:boosting-state}
  \For{$k=0$ to $K-1$}
    \State Prove leaf assignment correctness as in \cref{prop:leaf-assignment}
    \State Prove leaf histogram correctness by aggregation at addresses $\mathbf{addr}^{(t,k)}$ as in \cref{prop:leaf-hist}
    \State Prove internal propagation as in \cref{prop:internal-propagation}
    \State Prove prefix, total, and right-side statistics as in \cref{prop:prefix-total}
    \State Prove candidate split-score correctness via zero-safe quotient witnesses as in \cref{prop:candidate-score}
    \State Prove split optimality as in \cref{prop:split-optimality}
    \State Prove leaf values and round closure as in \cref{prop:leaf-values}
  \EndFor
  \State Conclude round-level correctness via \cref{thm:round-correctness}
\EndFor
\State Batch heterogeneous linear identities using domain-lifting batching
\State Batch aligned same-form nonlinear gadgets using interleaving batching
\State \Return the resulting proof transcript
\end{algorithmic}
\end{algorithm}
