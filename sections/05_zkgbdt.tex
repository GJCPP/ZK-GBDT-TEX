\section{ZK-GBDT}\label{sec:zkgbdt-protocol}

As outlined in \Cref{sec:technical-overview},
    we commit the forest by committing
    \begin{align}
        \mathcal{F} = (\mathbf{lc}, \mathbf{rc}, \mathbf{N}^{\leftarrow}, \mathbf{N}^{\rightarrow}, \mathbf{E}, \boldsymbol{\Theta}, \mathbf{tree}, \mathbf{root}, \mathbf{score}, \mathbf{is\_leaf}).
    \end{align}

\subsection{Proof Primitives}
\paragraph{Zero-check.}
Given $f(X)$ encoding vector $\mathbf{f}$ over $\mathbb{H}_n$,
    the zero-check validates that $\mathbf{f} = \mathbf{0}$ by proving the existence of $q(X)$ such that
    \begin{align}
        f(X) = q(X)\cdot \nu_n(X).
    \end{align}

We note that as noted in \Cref{thm:domain-lifting-batching}, all zero-checks can be batched together.
    
\paragraph{Sum-check.}
Given $f(X)$ encoding vector $\mathbf{f}$ over $\mathbb{H}_n$,
    the Aurora sum-check~\cite{aurora19} validates that $\sum_{i=0}^{n-1} \mathbf{f}[i] = s$ by proving the existence of $q(X)$
    and $r(X)$ such that
    \begin{align}
        \deg r(X) < n-1, \\
        f(X) - s/n = r(X) \cdot X + q(X)\cdot \nu_n(X).
    \end{align}
    
We note that as noted in \Cref{thm:domain-lifting-batching}, all sum-checks can be batched together.

\paragraph{Table lookup.}
Given $(\mathbf{F}_i)_{i\in [k]}$ and $(\mathbf{T}_i)_{i\in [k]}$
    encoding matrices $\mathbf{F}_i\in \mathbb{F}^{n\times m_i}$
    and $\mathbf{T}_i\in \mathbb{F}^{N\times m_i}$,
    the lookup relation validates that there exist
    $\mathbf{j}\in [N]^n$, such that
    \[
        \mathbf{F}_i[r, c] = \mathbf{T}_i[\mathbf{j}[r], c], \forall r\in [n], c\in [m_i], i\in [k].
    \]
    This is reduced to a logup instance~\cite{logup22},
    which is further batched together by 
    interleaving batching, c.f. \Cref{lem:interleaving}.
    See \Cref{app:batched-lookup}.

\paragraph{Quantization.}
Given $f(X)$ and $g(X)$ encoding vectors $\mathbf{f}$ and $\mathbf{g}$ over $\mathbb{H}_n$,
    and given a public quantization factor $q$,
    the quantization relation validates that
    \begin{align}
        \mathbf{f}[i] = \left\lfloor \frac{\mathbf{g}[i]}{2^q} \right\rfloor, \forall i\in [n].
    \end{align}
    This is reduced to proving the existence of $\mathbf{q}$ and $\mathbf{r}$ such that
    \begin{align}
        \mathbf{g}[i] = 2^q \cdot \mathbf{f}[i] + \mathbf{r}[i], \forall i\in [n], \\
        0 \leq \mathbf{r}[i] < 2^q, \forall i\in [n].
    \end{align}

As discussed earlier, we first batch all quantization checks
    by \Cref{lem:interleaving}.
    Then proving the range of $\mathbf{r}$ with lookup argument.

\paragraph{Prefix sum.}
Given $f(X)$ encoding vector $\mathbf{f}$ over $\mathbb{H}_{abc}$,
    we require three types of prefix sums:
    prove $\mathbf{g}$ and $\mathbf{h}$ such that
    \begin{enumerate}
        \item $\mathbf{g}[i, j, k] = \sum_{k'=0}^{k} \mathbf{f}[i, j, k']$ and $\mathbf{h}[i, j] = \sum_{k'=0}^{c} \mathbf{f}[i, j, k']$;
        \item $\mathbf{g}[i, j, k] = \sum_{j'=0}^{j} \mathbf{f}[i, j', k]$ and $\mathbf{h}[i, k] = \sum_{j'=0}^{b} \mathbf{f}[i, j', k]$;
        \item $\mathbf{g}[i, j, k] = \sum_{i'=0}^{i} \mathbf{f}[i', j, k]$ and $\mathbf{h}[j, k] = \sum_{i'=0}^{a} \mathbf{f}[i', j, k]$.
    \end{enumerate}
    We note that higher dimensional prefix sums can be reduced to 
    one of the three types,
    e.g. the prefix sum along the second dimension of
    $\mathbf{V}\in \mathbb{F}^{a\times b\times c\times d}$
    is reduced to proving the second type over
    $\mathbf{V}_{\{a\times b\times cd\}}$.
    
For brevity, we defer the concrete constructions to \Cref{app:prefix-sum}.

\paragraph{Column-replication.}
Given $f(X)$ encoding $\mathbf{f}\in \mathbb{F}^{n}$
    and $G(X)$ encoding $\mathbf{G}\in \mathbb{F}^{n\times m}$,
    the column-replication relation validates that 
    \begin{align}
        \forall i\in [n], j\in [m].\ \mathbf{G}[i][j] = \mathbf{f}[i].
    \end{align}
    We defer the concrete construction to \Cref{app:replication}.

\subsection{Proof of Well-formedness}

Following \Cref{subsec:proof of well-formedness},
    we prove the well-formedness of the committed forest by proving the following properties.

\paragraph{Tree index consistency.}
The prover commits $\mathbf{tree\_lc}$ and $\mathbf{tree\_rc}$ such that
    \begin{align}
        \mathbf{tree\_lc}[i] = \mathbf{tree}[\mathbf{lc}[i]], \\
        \mathbf{tree\_rc}[i] = \mathbf{tree}[\mathbf{rc}[i]],
    \end{align}
    and prove their correctness by proving the lookup relation
    \begin{align}
        \mathbf{lc}\| \mathbf{tree\_lc} \subseteq (0, 1, \dots, N_{\mathrm{tot}}-1)^\top \| \mathbf{tree},\\
        \mathbf{rc}\| \mathbf{tree\_rc} \subseteq (0, 1, \dots, N_{\mathrm{tot}}-1)^\top \| \mathbf{tree}.
    \end{align}
    The prover then uses zero-check over $\mathbb{H}_{N_{\mathrm{tot}}}$ to prove
    \begin{align}
        (\mathbf{1}-\mathbf{is\_leaf})\odot (\mathbf{tree\_lc}-\mathbf{tree}) = \mathbf{0}, \\
        (\mathbf{1}-\mathbf{is\_leaf})\odot (\mathbf{tree\_rc}-\mathbf{tree}) = \mathbf{0}.
    \end{align}

\paragraph{Consistency of leaf nodes.}
To prove that leaves have no child, the prover uses zero-check over $\mathbb{H}_{N_{\mathrm{tot}}}$ to validate
    \begin{align}
        \mathbf{is\_leaf} \odot \mathbf{lc} = \mathbf{0}, \\
        \mathbf{is\_leaf} \odot \mathbf{rc} = \mathbf{0}.
    \end{align}

\paragraph{Hyper-box of roots.}
To prove that the hyper-box of any root is full, the prover uses zero-check over $\mathbb{H}_{N_{\mathrm{tot}}d}$ to validate
    \begin{align}
        \mathbf{Root} \odot \mathbf{N}^{\leftarrow} = \mathbf{0}, \\
        \mathbf{Root} \odot \bigl(\mathbf{N}^{\rightarrow} - (B + 1)\mathbf{1}\bigr) = \mathbf{0}.
    \end{align}

\paragraph{Hyper-box consistency of children.}
To prove that the child hyper-boxes are consistent with the parent and the selected split,
    the prover first commits
    $\mathbf{N}^{\leftarrow}_{\mathrm{lc}}, \mathbf{N}^{\rightarrow}_{\mathrm{lc}},
    \mathbf{N}^{\leftarrow}_{\mathrm{rc}}, \mathbf{N}^{\rightarrow}_{\mathrm{rc}}
    \in \mathbb{F}^{N_{\mathrm{tot}}\times d}$
    and proves that
    \begin{align}
        \mathbf{lc}\| \mathbf{N}^{\leftarrow}_{\mathrm{lc}}\| \mathbf{N}^{\rightarrow}_{\mathrm{lc}}
        &\subseteq
        (0, 1, \dots, N_{\mathrm{tot}}-1)^\top\| \mathbf{N}^{\leftarrow}\| \mathbf{N}^{\rightarrow},\\
        \mathbf{rc}\| \mathbf{N}^{\leftarrow}_{\mathrm{rc}}\| \mathbf{N}^{\rightarrow}_{\mathrm{rc}}
        &\subseteq
        (0, 1, \dots, N_{\mathrm{tot}}-1)^\top\| \mathbf{N}^{\leftarrow}\| \mathbf{N}^{\rightarrow}.
    \end{align}
    Let $\mathbf{Int}\in\mathbb{F}^{N_{\mathrm{tot}}\times d}$ be generated by
    column-replication $d$ times from $\mathbf{1}-\mathbf{is\_leaf}$.
    The prover then uses zero-check over $\mathbb{H}_{N_{\mathrm{tot}}d}$ to validate
    \begin{align}
        \mathbf{Int}\odot\bigl(\mathbf{N}^{\leftarrow}-\mathbf{N}^{\leftarrow}_{\mathrm{lc}}\bigr) &= \mathbf{0},\\
        \mathbf{Int}\odot\bigl(\mathbf{N}^{\rightarrow}-\mathbf{N}^{\rightarrow}_{\mathrm{rc}}\bigr) &= \mathbf{0},\\
        \mathbf{Int}\odot\mathbf{E}\odot
        \bigl(\mathbf{N}^{\rightarrow}_{\mathrm{lc}}-\mathbf{N}^{\leftarrow}_{\mathrm{rc}}\bigr) &= \mathbf{0},\\
        \mathbf{Int}\odot(\mathbf{1}-\mathbf{E})\odot
        \bigl(\mathbf{N}^{\leftarrow}-\mathbf{N}^{\leftarrow}_{\mathrm{rc}}\bigr) &= \mathbf{0},\\
        \mathbf{Int}\odot(\mathbf{1}-\mathbf{E})\odot
        \bigl(\mathbf{N}^{\rightarrow}-\mathbf{N}^{\rightarrow}_{\mathrm{lc}}\bigr) &= \mathbf{0}.
    \end{align}

    Note that the above equations hold for dummy node $0$,
    since all its attributes are zero.


\paragraph{Feature-selection matrix.}
To prove that $\mathbf{E}$ is boolean and one-hot on internal nodes and zero on leaves,
    the prover first uses zero-check over $\mathbb{H}_{N_{\mathrm{tot}}d}$ to validate
    \begin{align}
        \mathbf{E}\odot(\mathbf{E}-\mathbf{1}) = \mathbf{0}.
    \end{align}
    It then invokes the prefix-sum primitive along the feature dimension to obtain
    $\mathbf{E}_{\mathrm{sum}}\in\mathbb{F}^{N_{\mathrm{tot}}}$, where
    \begin{align}
        \mathbf{E}_{\mathrm{sum}}[i]=\sum_{j=1}^{d}\mathbf{E}[i,j].
    \end{align}
    Finally, the prover uses zero-check over $\mathbb{H}_{N_{\mathrm{tot}}}$ to validate
    \begin{align}
        \mathbf{E}_{\mathrm{sum}}+\mathbf{is\_leaf}-\mathbf{1}=\mathbf{0}.
    \end{align}

\paragraph{Scores of internal nodes.}
To prove that internal nodes carry no leaf score, the prover uses zero-check over
    $\mathbb{H}_{N_{\mathrm{tot}}}$ to validate
    \begin{align}
        \mathbf{score}\odot(\mathbf{1}-\mathbf{is\_leaf})=\mathbf{0}.
    \end{align}

\paragraph{Number of roots.}
To prove that the committed forest contains exactly $TK$ trees,
    the prover first uses zero-check over $\mathbb{H}_{N_{\mathrm{tot}}}$ to validate
    the vector root is boolean,
    \begin{align}
        \mathbf{root}\odot(\mathbf{root}-\mathbf{1})=\mathbf{0}.
    \end{align}
    It then uses sum-check over $\mathbb{H}_{N_{\mathrm{tot}}}$ to prove
    \begin{align}
        \sum_{i=0}^{N_{\mathrm{tot}}-1}\mathbf{root}[i]=TK.
    \end{align}

\subsection{Proof of Inference}

Following \Cref{eq:inference-lookup-1}, the prover proves batched inference for
public inputs $\mathbf{I}\in [B]^{b\times d}$ and claimed one-hot labels
$\mathbf{Y}\in\{0, 1\}^{K\times b}$ by proving the following properties.

\paragraph{Leaf lookup.}
The prover commits
$\mathbf{U}^{\leftarrow},\mathbf{U}^{\rightarrow}\in\mathbb{F}^{TKb\times d}$
and $\mathbf{sc}\in\mathbb{F}^{TKb}$.
Let $\mathbf{ind\_tree}\in[TK+1]^{TKb}$ be the public vector
\begin{align}
    \mathbf{ind\_tree}_{\{T\times K\times b\}}[t,k,i] = tK+k+1,
\end{align}
consistent with the convention that non-dummy trees are indexed by
$1,\ldots,TK$.
The prover proves the lookup relation
\begin{align}
    \mathbf{1}\|\mathbf{ind\_tree}\|
    \mathbf{U}^{\leftarrow}\|\mathbf{U}^{\rightarrow}\|\mathbf{sc}
    \subseteq
    \mathbf{is\_leaf}\|\mathbf{tree}\|
    \mathbf{N}^{\leftarrow}\|\mathbf{N}^{\rightarrow}\|\mathbf{score}.
\end{align}
The leading all-one column forces each selected node to be a leaf, while the
tree-index column forces the selected leaf to belong to the intended tree.
Since the bounds and score are fetched in the same lookup row, this also binds
each row of $\mathbf{U}^{\leftarrow},\mathbf{U}^{\rightarrow}$ and
$\mathbf{sc}$ to the same committed leaf.

\paragraph{Interval containment.}
    Let $\mathbf{I}'\in\mathbb{F}^{TKb\times d}$ be the row-replication of $\mathbf{I}$ for $TK$ times, i.e. for each $t,k,i,j$,
    \begin{align}
        \mathbf{I}'_{\{T\times K\times b\times d\}}[t,k,i,j]
        =
        \mathbf{I}[i,j].
    \end{align}
    By \Cref{lem:row-replication}, we have
    $I'(X) = I(X^{TK})$,
    so the prover does not need to commit $\mathbf{I}'$.

To prove that every selected leaf contains the corresponding input sample, the
prover forms
\begin{align}
    \pmb{\Delta}^{\leftarrow}
    &=
    \mathbf{I}'-\mathbf{U}^{\leftarrow},\\
    \pmb{\Delta}^{\rightarrow}
    &=
    \mathbf{U}^{\rightarrow}-\mathbf{I}'-\mathbf{1},
\end{align}
and proves by range lookup over $\{0,\ldots,B\}$ that all entries of
$\pmb{\Delta}^{\leftarrow}$ and
$\pmb{\Delta}^{\rightarrow}$ are in range.
Equivalently, for every $t,k,i,j$,
\begin{align}
    \mathbf{U}^{\leftarrow}_{\{T\times K\times b\times d\}}[t,k,i,j]
    \leq
    \mathbf{I}[i,j]
    <
    \mathbf{U}^{\rightarrow}_{\{T\times K\times b\times d\}}[t,k,i,j].
\end{align}
Together with forest well-formedness, this proves that
$\mathbf{sc}_{\{T\times K\times b\}}[t,k,i]$ is the score assigned by the
tree $(t,k)$ to sample $i$.

\paragraph{Score aggregation.}
The prover commits
$\mathbf{S}\in\mathbb{F}^{K\times b}$, where
$\mathbf{S}[k,i]$ is the accumulated inference score of sample
$i$ for class $k$.
It invokes the prefix-sum primitive along the round dimension of
$\mathbf{sc}_{\{T\times K\times b\}}$ to prove
\begin{align}
    \mathbf{S}[k,i]
    =
    \sum_{t=0}^{T-1}
    \mathbf{sc}_{\{T\times K\times b\}}[t,k,i],
    \qquad
    \forall k\in[K], i\in[b].
\end{align}

\paragraph{Maximum selection.}
To prove that $\mathbf{Y}$ contains the predicted class labels, the verifier
first checks directly that $\mathbf{Y}$ is one-hot.
The prover commits a masked score matrix
$\widehat{\mathbf{S}}\in\mathbb{F}^{K\times b}$ and proves by zero-check
over $\mathbb{H}_{Kb}$ that
\begin{align}
    \widehat{\mathbf{S}}
    -
    \mathbf{Y}\odot\mathbf{S}
    =
    \mathbf{0}.
\end{align}

The prover then invokes the prefix-sum primitive along the class dimension to
obtain $\mathbf{s}\in\mathbb{F}^{b}$ and proves
\begin{align}
    \mathbf{s}[i]
    =
    \sum_{k=0}^{K-1}\widehat{\mathbf{S}}[k,i],
    \qquad
    \forall i\in[b].
\end{align}

Row-replicating $\mathbf{s}$ for $K$ times 
    to obtain $\mathbf{S}'\in \mathbb{F}^{K\times b}$, the prove finally proves
    \begin{align}
        0\leq \mathbf{S}' - \mathbf{S} < \mathcal{B},
    \end{align}
    which certifies that $\mathbf{S}'$ selects the maximum score in each column of $\mathbf{S}$, and thus certifies the correctness of $\mathbf{Y}$.

\subsection{Proof of Training}

The prover commits the training dataset
$\mathbf{D}\in[B]^{M\times d}$, the labels
$\mathbf{Y}\in\{0,1\}^{K\times M}$, the pseudo-residuals
$\mathbf{G}\in\mathbb{F}^{T\times K\times M}$, and
$\mathbf{Ind}\in[N_{\mathrm{tot}}]^{T\times K\times M}$
which indicates where a sample falls in each tree.
The proof then certifies each trained tree in the following order.

\paragraph{Dataset leaf assignment.}
The prover commits
$\mathbf{U}^{\leftarrow},\mathbf{U}^{\rightarrow}
\in\mathbb{F}^{TKM\times d}$.
Let
\begin{align}
    \mathbf{ind\_tree}_{\{T\times K\times M\}}[t,k,r]=tK+k+1,
\end{align}
which is column-replicating $(1, 2, \dots, TK)^\top$ for $M$ times.
The prover proves
\begin{align}
    \mathbf{Ind}_{\{TKM\}}\|\mathbf{1}\|\mathbf{ind\_tree}\|
    \mathbf{U}^{\leftarrow}\|\mathbf{U}^{\rightarrow}
    &\subseteq\\
    (0,1,\dots,N_{\mathrm{tot}}-1)^\top\|&
    \mathbf{is\_leaf}\|\mathbf{tree}\|\mathbf{N}^{\leftarrow}\|\mathbf{N}^{\rightarrow},
\end{align}
where $\mathbf{ind}\in [N_{\mathrm{tot}}]^M$ indicates the leaf to which each training sample falls.
It then row-replicates $\mathbf{D}$ for $TK$ times and proves by range lookup
that, for all $t,k,r,j$,
\begin{align}
    \mathbf{U}^{\leftarrow}_{\{T\times K\times M\times d\}}[t,k,r,j]
    \leq
    \mathbf{D}[r,j]
    <
    \mathbf{U}^{\rightarrow}_{\{T\times K\times M\times d\}}[t,k,r,j].
\end{align}
This proves that $\mathbf{Ind}[t,k,r]$ is the leaf reached by sample $r$ in
tree $(t,k)$.

\paragraph{Histogram.}
For the current tree $(t,k)$, the prover commits leaf histograms
$\mathbf{C}_{\mathrm{leaf}},\mathbf{V}_{\mathrm{leaf}}
\in\mathbb{F}^{N_{\mathrm{tot}}\times d\times B}$.
It also commits an address tensor
$\mathbf{A}\in\mathbb{F}^{T\times K\times M\times d}$
and proves its correctness by zero-check:
\begin{align}
    &\forall t\in[T],k\in[K],r\in[M],j\in[d]\nonumber\\
    &\mathbf{A}[t,k,r,j]
    -\mathbf{Ind}[t,k,r]\cdot dB+j\cdot B-\mathbf{D}[r,j]
    =
    0.
\end{align}
This check is aligned by column-replicating $\mathbf{Ind}$ and row-replicating $\mathbf{D}$ to $(TKMd)$-size in total.

For a random challenge $\alpha$, the prover commits and prove
the correctness of $\mathbf{P} = (1, \alpha, \dots, \alpha^{N_{\mathrm{tot}}dB-1})$ by proving
\begin{align}
    \left(1 - \lambda_{N_{\mathrm{tot}}dB}^{(0)}(X)\right) \left(P(X) - \alpha P(\omega^{-1}_{N_{\mathrm{tot}}dB}X)\right) = q(X)\cdot \nu_{N_{\mathrm{tot}}dB}(X),
\end{align}
where recall that $\lambda_{N_{\mathrm{tot}}dB}^{(0)}$ is the Lagrange polynomial that vanishes on all points in $\mathbb{H}_{N_{\mathrm{tot}}dB}$ except $0$.

The prover then creates $\mathbf{A}'[t,k,r,j] = \alpha^{\mathbf{A}[t,k,r,j]}$ and proves its correctness by lookup:
\begin{align}
    \mathbf{A}\|\mathbf{A}'
    &=
    (0, 1, \dots, N_{\mathrm{tot}}dB-1)^\top\|
    \mathbf{P}.
\end{align}
The prover then proves the correctness of the leaf histograms by sum-checking
\begin{align}
    \sum_z \mathbf{C}_{\mathrm{leaf},\{N_{\mathrm{tot}}dB\}}[z] \cdot \mathbf{P}[z]
    &=
    \sum_{r,j}\mathbf{A}'[t,k,r,j],\\
    \sum_z \mathbf{V}_{\mathrm{leaf},\{N_{\mathrm{tot}}dB\}}[z] \cdot \mathbf{P}[z]
    &=
    \sum_{r,j}\mathbf{A}'[t,k,r,j]\cdot \mathbf{G}[t,k,r].
\end{align}
The prover then commits full histograms
$\mathbf{C},\mathbf{V}
\in\mathbb{F}^{N_{\mathrm{tot}}\times d\times B}$.
It commits routed child histograms
$\mathbf{C}_{\mathrm{lc}},\mathbf{C}_{\mathrm{rc}},
\mathbf{V}_{\mathrm{lc}},\mathbf{V}_{\mathrm{rc}}
\in\mathbb{F}^{N_{\mathrm{tot}}\times d\times B}$,
and proves their correctness by lookup:
\begin{align}
    \mathbf{lc}\|\mathbf{C}_{\mathrm{lc}}\|\mathbf{V}_{\mathrm{lc}}
    &\subseteq
    (0,1,\dots,N_{\mathrm{tot}}-1)^\top\|
    \mathbf{C}\|\mathbf{V},\\
    \mathbf{rc}\|\mathbf{C}_{\mathrm{rc}}\|\mathbf{V}_{\mathrm{rc}}
    &\subseteq
    (0,1,\dots,N_{\mathrm{tot}}-1)^\top\|
    \mathbf{C}\|\mathbf{V}.
\end{align}
Let $\mathbf{Leaf}\in\mathbb{F}^{N_{\mathrm{tot}}\times d\times B}$
be generated by column-replication $dB$ times from $\mathbf{is\_leaf}$.
The prover then uses zero-check over $\mathbb{H}_{N_{\mathrm{tot}}dB}$ to validate
\begin{align}
    \mathbf{C}
    -
    \mathbf{Leaf}\odot\mathbf{C}_{\mathrm{leaf}}
    +(\mathbf{1}-\mathbf{Leaf})\odot
    \bigl(\mathbf{C}_{\mathrm{lc}}+\mathbf{C}_{\mathrm{rc}}\bigr) &= 0,\\
    \mathbf{V}
    -
    \mathbf{Leaf}\odot\mathbf{V}_{\mathrm{leaf}}
    +(\mathbf{1}-\mathbf{Leaf})\odot
    \bigl(\mathbf{V}_{\mathrm{lc}}+\mathbf{V}_{\mathrm{rc}}\bigr) &= 0.
\end{align}

\paragraph{Optimal split.}
The prover invokes the prefix-sum primitive along the bin dimension to obtain
$\mathbf{C}_{\leq},\mathbf{V}_{\leq}$ and
$\mathbf{C}_{>},\mathbf{V}_{>}$.
For every candidate $(u,j,b)$, the prover commits the gain tensor
$\boldsymbol{\Gamma}\in\mathbb{F}^{N_{\mathrm{tot}}\times d\times B}$
and a validity mask $\boldsymbol{\chi}$, and proves
\begin{align}
    \boldsymbol{\chi}\odot
    \bigl(
    \boldsymbol{\Gamma}\odot\mathbf{C}_{\leq}\odot\mathbf{C}_{>}
    -(\mathbf{V}_{\leq})^2\odot\mathbf{C}_{>}
    -(\mathbf{V}_{>})^2\odot\mathbf{C}_{\leq}
    \bigr)
    &=\mathbf{0},\\
    (\mathbf{1}-\boldsymbol{\chi})\odot\boldsymbol{\Gamma}
    &=\mathbf{0},
\end{align}
where $\boldsymbol{\chi}=1$ only for candidates with both children nonempty,
and its correctness can be proven by lookup.
The prover commits $\boldsymbol{\Gamma}^{*}\in
\mathbb{F}^{N_{\mathrm{tot}}}$ and a one-hot threshold mask
$\mathbf{T}\in \{0, 1\}^{N_{\mathrm{tot}}\times B}$.
The prover proves $\mathbf{T}$ by booleanity and row-sum checks, and proves
by prefix sums over the feature and threshold dimensions that
\begin{align}
    \forall u\in [N_{\mathrm{tot}}],\ 
    \boldsymbol{\Gamma}^{*}[u]
    =
    \sum_{j=1}^{d}\sum_{b=0}^{B-1}
    \mathbf{E}[u,j]\mathbf{T}[u,b]\boldsymbol{\Gamma}[u,j,b].
\end{align}
Finally, it proves by range lookup that
\begin{align}
    0\leq
    \boldsymbol{\Gamma}^{*}[u]
    -
    \boldsymbol{\Gamma}[u,j,b]
    <\mathcal{B}
\end{align}
for every valid candidate $(j,b)$.

\paragraph{Applying the selected split.}
The prover links the one-hot threshold mask $\mathbf{T}$ to the committed threshold vector 
by zero-checking
\begin{align}
    (\mathbf{1}-\mathbf{is\_leaf})\odot
    \left(
    \boldsymbol{\Theta}
    -
    \sum_{b=0}^{B-1} b\cdot\mathbf{T}[\cdot,b]
    \right)
    &=\mathbf{0}.
\end{align}
Using the routed child bounds already committed in the well-formedness proof,
the prover then validates
\begin{align}
    \mathbf{Int}\odot\mathbf{E}\odot
    \bigl(\mathbf{N}^{\rightarrow}_{\mathrm{lc}}-\boldsymbol{\Theta}'\bigr)
    &=\mathbf{0},\\
    \mathbf{Int}\odot\mathbf{E}\odot
    \bigl(\mathbf{N}^{\leftarrow}_{\mathrm{rc}}-\boldsymbol{\Theta}'\bigr)
    &=\mathbf{0},
\end{align}
where $\boldsymbol{\Theta}'\in\mathbb{F}^{N_{\mathrm{tot}}\times d}$ is
generated by column-replication from $\boldsymbol{\Theta}$.
Together with the maximum proof above and the hyper-box consistency checks,
this proves that the optimal split selected from the histograms is exactly the
split applied in the committed forest.
